  <script> 

    const STORAGE_KEY = 'smartLibraryBooks_v2'; 

    const bookForm = document.getElementById('bookForm'); 

    const editId = document.getElementById('editId'); 

    const title = document.getElementById('title'); 

    const author = document.getElementById('author'); 

    const category = document.getElementById('category'); 

    const member = document.getElementById('member'); 

    const issueDate = document.getElementById('issueDate'); 

    const returnDate = document.getElementById('returnDate'); 

    const status = document.getElementById('status'); 

    const saveBtn = document.getElementById('saveBtn'); 

    const resetBtn = document.getElementById('resetBtn'); 

    const searchInput = document.getElementById('searchInput'); 

    const filterCategory = document.getElementById('filterCategory'); 

    const filterStatus = document.getElementById('filterStatus'); 

    const filterDate = document.getElementById('filterDate'); 

    const bookList = document.getElementById('bookList'); 

    const totalBooks = document.getElementById('totalBooks'); 

    const availableBooks = document.getElementById('availableBooks'); 

    const issuedBooks = document.getElementById('issuedBooks'); 

    const overdueBooks = document.getElementById('overdueBooks'); 

    const clearAllBtn = document.getElementById('clearAllBtn'); 

    const printBtn = document.getElementById('printBtn'); 

    const clockText = document.getElementById('clockText'); 

 

    let books = JSON.parse(localStorage.getItem(STORAGE_KEY) || 'null') || [ 

      { id: 1, title: 'Physics Fundamentals', author: 'R. Sharma', category: 'Science', member: '', issueDate: '', returnDate: '', status: 'Available' }, 

      { id: 2, title: 'Modern Mathematics', author: 'S. Verma', category: 'Mathematics', member: 'Aman', issueDate: '2026-04-28', returnDate: '2026-05-05', status: 'Issued' }, 

      { id: 3, title: 'JavaScript Basics', author: 'A. Mehta', category: 'Computer', member: '', issueDate: '', returnDate: '', status: 'Available' }, 

      { id: 4, title: 'India Through Ages', author: 'P. Joshi', category: 'History', member: 'Riya', issueDate: '2026-04-20', returnDate: '2026-04-26', status: 'Issued' }, 

      { id: 5, title: 'Moral Stories', author: 'K. Gupta', category: 'Story', member: '', issueDate: '', returnDate: '', status: 'Available' } 

    ]; 

 

    function save() { localStorage.setItem(STORAGE_KEY, JSON.stringify(books)); } 

    function todayISO() { return new Date().toISOString().split('T')[0]; } 

    function isOverdue(book) { return book.status === 'Issued' && book.returnDate && book.returnDate < todayISO(); } 

    function updateClock() { const d = new Date(); clockText.textContent = d.toLocaleString('en-IN', { dateStyle: 'medium', timeStyle: 'medium' }); } 

    updateClock(); setInterval(updateClock, 1000); 

 

    function fillCategoryFilter() { 

      const prev = filterCategory.value; 

      const cats = [...new Set(books.map(b => b.category))].sort(); 

      filterCategory.innerHTML = '<option value="All">All Categories</option>' + cats.map(c => `<option value="${c}">${c}</option>`).join(''); 

      filterCategory.value = cats.includes(prev) ? prev : 'All'; 

    } 

 

    function render() { 

      save(); 

      fillCategoryFilter(); 

      const q = searchInput.value.trim().toLowerCase(); 

      const c = filterCategory.value; 

      const s = filterStatus.value; 

      const d = filterDate.value; 

      const filtered = books.filter(b => { 

        const okSearch = !q || [b.title, b.author, b.member].join(' ').toLowerCase().includes(q); 

        const okCategory = c === 'All' || b.category === c; 

        const okStatus = s === 'All' || b.status === s; 

        const okDate = !d || b.issueDate === d || b.returnDate === d; 

        return okSearch && okCategory && okStatus && okDate; 

      }); 

      totalBooks.textContent = books.length; 

      availableBooks.textContent = books.filter(b => b.status === 'Available').length; 

      issuedBooks.textContent = books.filter(b => b.status === 'Issued').length; 

      overdueBooks.textContent = books.filter(isOverdue).length; 

      if (!filtered.length) { 

        bookList.innerHTML = `<div class="empty"><h4>No records found</h4><p>Try changing the search or filter values.</p></div>`; 

        return; 

      } 

      bookList.innerHTML = filtered.map(book => { 

        const overdue = isOverdue(book); 

        return ` 

          <article class="book"> 

            <div> 

              <h4>${book.title}</h4> 

              <div>Author: ${book.author}</div> 

              <div class="meta"> 

                <span class="pill">${book.category}</span> 

                <span class="pill ${book.status === 'Available' ? 'ok' : 'bad'}">${book.status}</span> 

                ${overdue ? '<span class="pill warn">Overdue</span>' : ''} 

              </div> 

              <div class="small">Member: ${book.member || '-'}</div> 

              <div class="small">Issue Date: ${book.issueDate || '-'}</div> 

              <div class="small">Return Date: ${book.returnDate || '-'}</div> 

              <div class="small">Book ID: LIB-${String(book.id).padStart(3, '0')}</div> 

            </div> 

            <div class="actions"> 

              <button class="btn-primary" onclick="toggleStatus(${book.id})">${book.status === 'Available' ? 'Issue' : 'Return'}</button> 

              <button class="btn-soft" onclick="editBook(${book.id})">Edit</button> 

              <button class="btn-danger" onclick="removeBook(${book.id})">Delete</button> 

            </div> 

          </article> 

        `; 

      }).join(''); 

    } 

 

    bookForm.addEventListener('submit', e => { 

      e.preventDefault(); 

      const data = { title: title.value.trim(), author: author.value.trim(), category: category.value, member: member.value.trim(), issueDate: issueDate.value, returnDate: returnDate.value, status: status.value }; 

      if (!data.title || !data.author || !data.category) return; 

      if (editId.value) { 

        const idx = books.findIndex(b => String(b.id) === editId.value); 

        if (idx !== -1) books[idx] = { ...books[idx], ...data }; 

        editId.value = ''; 

        saveBtn.textContent = 'Add Book'; 

      } else { 

        books.unshift({ id: Date.now(), ...data }); 

      } 

      bookForm.reset(); 

      status.value = 'Available'; 

      render(); 

    }); 

 

    function toggleStatus(id) { books = books.map(b => b.id === id ? { ...b, status: b.status === 'Available' ? 'Issued' : 'Available' } : b); render(); } 

    function editBook(id) { 

      const b = books.find(x => x.id === id); 

      if (!b) return; 

      editId.value = String(b.id); 

      title.value = b.title; author.value = b.author; category.value = b.category; member.value = b.member || ''; issueDate.value = b.issueDate || ''; returnDate.value = b.returnDate || ''; status.value = b.status; saveBtn.textContent = 'Update Book'; window.scrollTo({ top: 0, behavior: 'smooth' }); 

    } 

    function removeBook(id) { if (!confirm('Delete this record?')) return; books = books.filter(b => b.id !== id); if (String(editId.value) === String(id)) resetForm(); render(); } 

    function resetForm() { bookForm.reset(); editId.value = ''; status.value = 'Available'; saveBtn.textContent = 'Add Book'; } 

 

    resetBtn.addEventListener('click', resetForm); 

    searchInput.addEventListener('input', render); 

    filterCategory.addEventListener('change', render); 

    filterStatus.addEventListener('change', render); 

    filterDate.addEventListener('change', render); 

    clearAllBtn.addEventListener('click', () => { if (!confirm('Clear all library records?')) return; books = []; localStorage.removeItem(STORAGE_KEY); resetForm(); render(); }); 

    printBtn.addEventListener('click', () => window.print()); 

 

    render(); 

  </script>
